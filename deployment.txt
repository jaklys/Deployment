<script>
// Funkce pro načtení mapy z externího souboru
function loadHostMap() {
  fetch('map.json')
    .then(response => response.json())
    .then(data => {
      var hostname = "GPRTS0099883"; // Tuto hodnotu naplníte dynamicky
      var dns = "Nenalezeno";
      var env = "Nenalezeno";

      if (data.hasOwnProperty(hostname)) {
        dns = data[hostname].dns;
        env = data[hostname].env;
      }

      // Vložení hodnot do HTML
      document.title = dns + " - " + env;
      document.getElementById("dns").textContent = dns;
      document.getElementById("env").textContent = env;
    })
    .catch(error => console.error('Chyba při načítání mapy:', error));
}

// Volání funkce po načtení stránky
document.addEventListener("DOMContentLoaded", loadHostMap);
</script>


<h1 id="dns">Default DNS</h1>
<p id="env">Default ENV</p>
